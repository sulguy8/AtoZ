ui_num INT(11) NOT NULL AUTO_INCREMENT PRIMARY KEY ,

credat CHAR(8) NOT NULL,
cretim CHAR(6) NOT NULL,

modtim CHAR(6) NOT NULL,
active CHAR(1) DEFAULT '1'
);
