select bi_num, bi_title, bi_user, bi_content,
to_char(to_date(bi_credat),'YYYY-MM-DD') as bi_credat,
bi_cretim,
to_char(to_date(bi_moddat),'YYYY-MM-DD') as bi_moddat,
bi_modtim
from board_info
where bi_num = 5;
